==SYNTACTICAL==

dfile -> stmt*

stmt -> STR "=" expr

expr -> type_kw "{" ((table_elem,) | table_elem)* "}" | prim_stmt

elem        -> prim_stmt | enum_elem
enum_elem   -> STR ( "=" number )
struct_elem -> STR ":" expr

prim_stmt -> STR | str_lit | number

==LEXICAL==

type_kw -> "table" | "struct" | "enum"

str_lit -> '"' STR '"'
number -> ("0" ("x" | "o" | "b" ))? NUMSTR

STR = string of any number of ascii characters
NUMSTR = string of any number of numeric ascii characters
